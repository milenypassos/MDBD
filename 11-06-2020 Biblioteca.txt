create database biblioteca;

use biblioteca;

CREATE TABLE livro (
codLivro int PRIMARY KEY auto_increment,
titulo varchar(150),
autor varchar(150),
genero varchar(150),
editora varchar(150)
);

CREATE TABLE bibliotecario (
codFuncionario int PRIMARY KEY auto_increment,
nome varchar(150),
rg varchar(150) unique,
dataNascimento date,
email varchar(150) unique,
logradouro varchar(150),
num int,
complemento varchar(150),
bairro varchar(150),
cidade varchar(150),
uf varchar(2),
cep int,
cpf varchar(150) unique,
telefone varchar(150)
);


CREATE TABLE aluno (
ra int PRIMARY KEY auto_increment,
telefone varchar(15),
nome varchar(150),
turma varchar(150),
email varchar(150) unique,
dataNascimento date
);

CREATE TABLE emprestimo_devolucao (
codEmprestimo int PRIMARY KEY auto_increment,
data_emprestimo date,
data_devolucao date,
ra int,
codLivro int,
codFuncionario int,
FOREIGN KEY(ra) REFERENCES aluno (ra),
FOREIGN KEY(codLivro) REFERENCES livro (codLivro),
FOREIGN KEY(codFuncionario) REFERENCES bibliotecario (codFuncionario)
);
